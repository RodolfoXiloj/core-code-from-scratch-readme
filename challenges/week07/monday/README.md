<h3>String: substr() 📝</h3>

```js
function firstWord(string) {
	return string.substr(0, string.indexOf(' '));
}
```

}

<h3>Fahrenheit 📝</h3>

```js
function toFahrenheit(celcius) {
	return (celcius * 9/5) + 32;
}
```
